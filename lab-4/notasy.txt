double PI = 3.14159;


void ASK(double A1, double A2, double fn, double t, double M, std::vector<double>& out, int n) {

    if (n == 0) {
        for (int m = 0; m < M; m++) {
            out.push_back(A1 * sin(2 * PI * fn * t));
        }

    }

    else {
        for (int m = 0; m < M; m++) {
            out.push_back(A2 * sin(2 * PI * fn * t));
        }

    }


}
double PSK(double fn, double m, double t, double kp) {
    double zp = cos(2 * PI * fn * t + kp * m);
    return zp;
}
double FSK(double fn, double t, double fm, double m, double kf) {
    double zf = cos(2 * PI * fn * t + kf / fm * m);
    return zf;
}
